58 void SolenoidBase::Set(UINT8 value, UINT8 mask)

60  tRioStatusCode localStatus = NiFpga_Status_Success;

63  Synchronized sync(m_semaphore);

65  // Zero out the values to change
66  currentValue = currentValue & ~mask;
