22 void AnalogChannel::InitChannel(UINT8 moduleNumber, UINT32 channel)
23 {
24  char buf[64];
25  Resource::CreateResourceObject(&channels, kAnalogModules * kAnalogChannels);
26  if (!CheckAnalogModule(moduleNumber))
27  {
28  snprintf(buf, 64, "Analog Module %d", moduleNumber);
29  wpi_setWPIErrorWithContext(ModuleIndexOutOfRange, buf);
30  return;
31  }
32  if (!CheckAnalogChannel(channel))
33  {
34  snprintf(buf, 64, "Analog Channel %d", channel);
35  wpi_setWPIErrorWithContext(ChannelIndexOutOfRange, buf);
36  return;
37  }
38 
39  snprintf(buf, 64, "Analog Input %d (Module: %d)", channel, moduleNumber);
40  if (channels->Allocate((moduleNumber - 1) * kAnalogChannels + channel - 1, buf) == ~0ul)
41  {
42  CloneError(channels);
43  return;
44  }
45  m_channel = channel;
46  m_module = AnalogModule::GetInstance(moduleNumber);
47  if (IsAccumulatorChannel())
48  {
49  tRioStatusCode localStatus = NiFpga_Status_Success;
50  m_accumulator = tAccumulator::create(channel - 1, &localStatus);
51  wpi_setError(localStatus);
52  m_accumulatorOffset=0;
53  }
54  else
55  {
56  m_accumulator = NULL;
57  }
58 
59  nUsageReporting::report(nUsageReporting::kResourceType_AnalogChannel, channel, GetModuleNumber() - 1);
60 }

261 bool AnalogChannel::IsAccumulatorChannel()
262 {
263  if (StatusIsFatal()) return false;
264  if(m_module->GetNumber() != kAccumulatorModuleNumber) return false;
265  for (UINT32 i=0; i<kAccumulatorNumChannels; i++)
266  {
267  if (m_channel == kAccumulatorChannels[i]) return true;
268  }
269  return false;
270 }
