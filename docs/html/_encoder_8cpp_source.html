25 void Encoder::InitEncoder(bool reverseDirection, EncodingType encodingType)
26 {
27  m_encodingType = encodingType;
28  tRioStatusCode localStatus = NiFpga_Status_Success;
29  switch (encodingType)
30  {
31  case k4X:
32  Resource::CreateResourceObject(&quadEncoders, tEncoder::kNumSystems);
33  UINT32 index = quadEncoders->Allocate("4X Encoder");
34  if (index == ~0ul)
35  {
36  CloneError(quadEncoders);
37  return;
38  }
39  if (m_aSource->StatusIsFatal())
40  {
41  CloneError(m_aSource);
42  return;
43  }
44  if (m_bSource->StatusIsFatal())
45  {
46  CloneError(m_bSource);
47  return;
48  }
49  m_index = index;
50  m_encoder = tEncoder::create(m_index, &localStatus);
51  m_encoder->writeConfig_ASource_Module(m_aSource->GetModuleForRouting(), &localStatus);
52  m_encoder->writeConfig_ASource_Channel(m_aSource->GetChannelForRouting(), &localStatus);
53  m_encoder->writeConfig_ASource_AnalogTrigger(m_aSource->GetAnalogTriggerForRouting(), &localStatus);
54  m_encoder->writeConfig_BSource_Module(m_bSource->GetModuleForRouting(), &localStatus);
55  m_encoder->writeConfig_BSource_Channel(m_bSource->GetChannelForRouting(), &localStatus);
56  m_encoder->writeConfig_BSource_AnalogTrigger(m_bSource->GetAnalogTriggerForRouting(), &localStatus);
57  m_encoder->strobeReset(&localStatus);
58  m_encoder->writeConfig_Reverse(reverseDirection, &localStatus);
59  m_encoder->writeTimerConfig_AverageSize(4, &localStatus);
60  m_counter = NULL;
61  break;
62  case k1X:
63  case k2X:
64  m_counter = new Counter(m_encodingType, m_aSource, m_bSource, reverseDirection);
65  m_index = m_counter->GetIndex();
66  break;
67  }
68  m_distancePerPulse = 1.0;
69  m_pidSource = kDistance;
70  wpi_setError(localStatus);
71 
72  nUsageReporting::report(nUsageReporting::kResourceType_Encoder, m_index, encodingType);
73 }

294 double Encoder::GetPeriod()
295 {
296  if (StatusIsFatal()) return 0.0;
297  double measuredPeriod;
298  if (m_counter)
299  {
300  measuredPeriod = m_counter->GetPeriod();
301  }
302  else
303  {
304  tRioStatusCode localStatus = NiFpga_Status_Success;
305  tEncoder::tTimerOutput output = m_encoder->readTimerOutput(&localStatus);
306  double value;
307  if (output.Stalled)
308  {
309  // Return infinity
310  double zero = 0.0;
311  value = 1.0 / zero;
312  }
313  else
314  {
315  // output.Period is a fixed point number that counts by 2 (24 bits, 25 integer bits)
316  value = (double)(output.Period << 1) / (double)output.Count;
317  }
318  wpi_setError(localStatus);
319  measuredPeriod = value * 1.0e-6;
320  }
321  return measuredPeriod / DecodingScaleFactor();
322 }
