31 void CANJaguar::InitCANJaguar()
32 {
33  m_transactionSemaphore = semMCreate(SEM_Q_PRIORITY | SEM_INVERSION_SAFE | SEM_DELETE_SAFE);
34  if (m_deviceNumber < 1 || m_deviceNumber > 63)
35  {
36  char buf[256];
37  snprintf(buf, 256, "device number \"%d\" must be between 1 and 63", m_deviceNumber);
38  wpi_setWPIErrorWithContext(ParameterOutOfRange, buf);
39  return;
40  }
41  UINT32 fwVer = GetFirmwareVersion();
42  if (StatusIsFatal())
43  return;
44  // 3330 was the first shipping RDK firmware version for the Jaguar
45  if (fwVer >= 3330 || fwVer < 92)
46  {
47  char buf[256];
48  if (fwVer < 3330)
49  {
50  snprintf(buf, 256, "Jag #%d firmware (%d) is too old (must be at least version 92 of the FIRST approved firmware)", m_deviceNumber, fwVer);
51  }
52  else
53  {
54  snprintf(buf, 256, "Jag #%d firmware (%d) is not FIRST approved (must be at least version 92 of the FIRST approved firmware)", m_deviceNumber, fwVer);
55  }
56  wpi_setWPIErrorWithContext(JaguarVersionError, buf);
57  return;
58  }
59  switch (m_controlMode)
60  {
61  case kPercentVbus:
62  case kVoltage:
63  // No additional configuration required... start enabled.
64  EnableControl();
65  break;
66  default:
67  break;
68  }
69  m_safetyHelper = new MotorSafetyHelper(this);
70 
71  nUsageReporting::report(nUsageReporting::kResourceType_CANJaguar, m_deviceNumber, m_controlMode);
72 }

110 void CANJaguar::Set(float outputValue, UINT8 syncGroup)
111 {
112  UINT32 messageID;
113  UINT8 dataBuffer[8];
114  UINT8 dataSize;
115 
116  if (m_safetyHelper && !m_safetyHelper->IsAlive())
117  {
118  EnableControl();
119  }
120 
121  switch(m_controlMode)
122  {
123  case kPercentVbus:
124  {
125  messageID = LM_API_VOLT_T_SET;
126  if (outputValue > 1.0) outputValue = 1.0;
127  if (outputValue < -1.0) outputValue = -1.0;
128  dataSize = packPercentage(dataBuffer, outputValue);
129  }
130  break;
131  case kSpeed:
132  {
133  messageID = LM_API_SPD_T_SET;
134  dataSize = packFXP16_16(dataBuffer, outputValue);
135  }
136  break;
137  case kPosition:
138  {
139  messageID = LM_API_POS_T_SET;
140  dataSize = packFXP16_16(dataBuffer, outputValue);
141  }
142  break;
143  case kCurrent:
144  {
145  messageID = LM_API_ICTRL_T_SET;
146  dataSize = packFXP8_8(dataBuffer, outputValue);
147  }
148  break;
149  case kVoltage:
150  {
151  messageID = LM_API_VCOMP_T_SET;
152  dataSize = packFXP8_8(dataBuffer, outputValue);
153  }
154  break;
155  default:
156  return;
157  }
158  if (syncGroup != 0)
159  {
160  dataBuffer[dataSize] = syncGroup;
161  dataSize++;
162  }
163  setTransaction(messageID, dataBuffer, dataSize);
164  if (m_safetyHelper) m_safetyHelper->Feed();
165 }

331 INT32 CANJaguar::sendMessage(UINT32 messageID, const UINT8 *data, UINT8 dataSize)
332 {
333  static const UINT32 kTrustedMessages[] = {
334  LM_API_VOLT_T_EN, LM_API_VOLT_T_SET, LM_API_SPD_T_EN, LM_API_SPD_T_SET,
335  LM_API_VCOMP_T_EN, LM_API_VCOMP_T_SET, LM_API_POS_T_EN, LM_API_POS_T_SET,
336  LM_API_ICTRL_T_EN, LM_API_ICTRL_T_SET};
337  INT32 status=0;
338 
339  for (UINT8 i=0; i<(sizeof(kTrustedMessages)/sizeof(kTrustedMessages[0])); i++)
340  {
341  if ((kFullMessageIDMask & messageID) == kTrustedMessages[i])
342  {
343  UINT8 dataBuffer[8];
344  dataBuffer[0] = 0;
345  dataBuffer[1] = 0;
346  // Make sure the data will still fit after adjusting for the token.
347  if (dataSize > 6)
348  {
349  // TODO: I would rather this not have to set the global error
350  wpi_setGlobalWPIErrorWithContext(ParameterOutOfRange, "dataSize > 6");
351  return 0;
352  }
353  for (UINT8 j=0; j < dataSize; j++)
354  {
355  dataBuffer[j + 2] = data[j];
356  }
357  FRC_NetworkCommunication_JaguarCANDriver_sendMessage(messageID, dataBuffer, dataSize + 2, &status);
358  return status;
359  }
360  }
361  FRC_NetworkCommunication_JaguarCANDriver_sendMessage(messageID, data, dataSize, &status);
362  return status;
363 }

392 void CANJaguar::setTransaction(UINT32 messageID, const UINT8 *data, UINT8 dataSize)
393 {
394  UINT32 ackMessageID = LM_API_ACK | m_deviceNumber;
395  INT32 localStatus = 0;
396 
397  // If there was an error on this object and it wasn't a timeout, refuse to talk to the device
398  // Call ClearError() on the object to try again
399  if (StatusIsFatal() && GetError().GetCode() != -44087)
400  return;
401 
402  // Make sure we don't have more than one transaction with the same Jaguar outstanding.
403  semTake(m_transactionSemaphore, WAIT_FOREVER);
404 
405  // Throw away any stale acks.
406  receiveMessage(&ackMessageID, NULL, 0, 0.0f);
407  // Send the message with the data.
408  localStatus = sendMessage(messageID | m_deviceNumber, data, dataSize);
409  wpi_setErrorWithContext(localStatus, "sendMessage");
410  // Wait for an ack.
411  localStatus = receiveMessage(&ackMessageID, NULL, 0);
412  wpi_setErrorWithContext(localStatus, "receiveMessage");
413 
414  // Transaction complete.
415  semGive(m_transactionSemaphore);
416 }

427 void CANJaguar::getTransaction(UINT32 messageID, UINT8 *data, UINT8 *dataSize)
428 {
429  UINT32 targetedMessageID = messageID | m_deviceNumber;
430  INT32 localStatus = 0;
431 
432  // If there was an error on this object and it wasn't a timeout, refuse to talk to the device
433  // Call ClearError() on the object to try again
434  if (StatusIsFatal() && GetError().GetCode() != -44087)
435  {
436  if (dataSize != NULL)
437  *dataSize = 0;
438  return;
439  }
440 
441  // Make sure we don't have more than one transaction with the same Jaguar outstanding.
442  semTake(m_transactionSemaphore, WAIT_FOREVER);
443 
444  // Throw away any stale responses.
445  receiveMessage(&targetedMessageID, NULL, 0, 0.0f);
446  // Send the message requesting data.
447  localStatus = sendMessage(targetedMessageID, NULL, 0);
448  wpi_setErrorWithContext(localStatus, "sendMessage");
449  // Caller may have set bit31 for remote frame transmission so clear invalid bits[31-29]
450  targetedMessageID &= 0x1FFFFFFF;
451  // Wait for the data.
452  localStatus = receiveMessage(&targetedMessageID, data, dataSize);
453  wpi_setErrorWithContext(localStatus, "receiveMessage");
454 
455  // Transaction complete.
456  semGive(m_transactionSemaphore);
457 }
