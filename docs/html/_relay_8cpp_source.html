25 void Relay::InitRelay (UINT8 moduleNumber)
26 {
27  char buf[64];
28  Resource::CreateResourceObject(&relayChannels, tDIO::kNumSystems * kRelayChannels * 2);
29  if (!SensorBase::CheckRelayModule(moduleNumber))
30  {
31  snprintf(buf, 64, "Digital Module %d", moduleNumber);
32  wpi_setWPIErrorWithContext(ModuleIndexOutOfRange, buf);
33  return;
34  }
35  if (!SensorBase::CheckRelayChannel(m_channel))
36  {
37  snprintf(buf, 64, "Relay Channel %d", m_channel);
38  wpi_setWPIErrorWithContext(ChannelIndexOutOfRange, buf);
39  return;
40  }
41 
42  if (m_direction == kBothDirections || m_direction == kForwardOnly)
43  {
44  snprintf(buf, 64, "Forward Relay %d (Module: %d)", m_channel, moduleNumber);
45  if (relayChannels->Allocate(((moduleNumber - 1) * kRelayChannels + m_channel - 1) * 2, buf) == ~0ul)
46  {
47  CloneError(relayChannels);
48  return;
49  }
50 
51  nUsageReporting::report(nUsageReporting::kResourceType_Relay, m_channel, moduleNumber - 1);
52  }
53  if (m_direction == kBothDirections || m_direction == kReverseOnly)
54  {
55  snprintf(buf, 64, "Reverse Relay %d (Module: %d)", m_channel, moduleNumber);
56  if (relayChannels->Allocate(((moduleNumber - 1) * kRelayChannels + m_channel - 1) * 2 + 1, buf) == ~0ul)
57  {
58  CloneError(relayChannels);
59  return;
60  }
61 
62  nUsageReporting::report(nUsageReporting::kResourceType_Relay, m_channel + 128, moduleNumber - 1);
63  }
64  m_module = DigitalModule::GetInstance(moduleNumber);
65  m_module->SetRelayForward(m_channel, false);
66  m_module->SetRelayReverse(m_channel, false);
67 }

128 void Relay::Set(Relay::Value value)
129 {
130  if (StatusIsFatal()) return;
131  switch (value)
132  {
133  case kOff:
134  if (m_direction == kBothDirections || m_direction == kForwardOnly)
135  {
136  m_module->SetRelayForward(m_channel, false);
137  }
138  if (m_direction == kBothDirections || m_direction == kReverseOnly)
139  {
140  m_module->SetRelayReverse(m_channel, false);
141  }
142  break;
143  case kOn:
144  if (m_direction == kBothDirections || m_direction == kForwardOnly)
145  {
146  m_module->SetRelayForward(m_channel, true);
147  }
148  if (m_direction == kBothDirections || m_direction == kReverseOnly)
149  {
150  m_module->SetRelayReverse(m_channel, true);
151  }
152  break;
153  case kForward:
154  if (m_direction == kReverseOnly)
155  {
156  wpi_setWPIError(IncompatibleMode);
157  break;
158  }
159  if (m_direction == kBothDirections || m_direction == kForwardOnly)
160  {
161  m_module->SetRelayForward(m_channel, true);
162  }
163  if (m_direction == kBothDirections)
164  {
165  m_module->SetRelayReverse(m_channel, false);
166  }
167  break;
168  case kReverse:
169  if (m_direction == kForwardOnly)
170  {
171  wpi_setWPIError(IncompatibleMode);
172  break;
173  }
174  if (m_direction == kBothDirections)
175  {
176  m_module->SetRelayForward(m_channel, false);
177  }
178  if (m_direction == kBothDirections || m_direction == kReverseOnly)
179  {
180  m_module->SetRelayReverse(m_channel, true);
181  }
182  break;
183  }
184 }
